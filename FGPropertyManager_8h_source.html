436  template <typename T> void
437  Tie (const std::string &name, T *pointer)
438  {
439  SGPropertyNode* property = root->getNode(name.c_str(), true);
440  if (!property) {
441  cerr << "Could not get or create property " << name << endl;
442  return;
443  }
444 
445  if (!property->tie(SGRawValuePointer<T>(pointer), false))
446  cerr << "Failed to tie property " << name << " to a pointer" << endl;
447  else {
448  tied_properties.push_back(property);
449  if (FGJSBBase::debug_lvl & 0x20) cout << name << endl;
450  }
451  }
452 
467  template <typename T> void
468  Tie (const std::string &name, T (*getter)(), void (*setter)(T) = nullptr)
469  {
470  SGPropertyNode* property = root->getNode(name.c_str(), true);
471  if (!property) {
472  std::cerr << "Could not get or create property " << name << std::endl;
473  return;
474  }
475 
476  if (!property->tie(SGRawValueFunctions<T>(getter, setter), false))
477  std::cerr << "Failed to tie property " << name << " to functions"
478  << std::endl;
479  else {
480  if (!setter) property->setAttribute(SGPropertyNode::WRITE, false);
481  if (!getter) property->setAttribute(SGPropertyNode::READ, false);
482  tied_properties.push_back(property);
483  if (FGJSBBase::debug_lvl & 0x20) std::cout << name << std::endl;
484  }
485  }
